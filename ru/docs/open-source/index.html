    Open Source

    Open Source #

    Econumo состоит из фронтенда и бэкенда.

    Бэкенд #

    Бэкенд распространяется с открытым исходным кодом под лицензией AGPLv3 и доступен на Github. Он написан на PHP и использует фреймворк Symfony. Docker-образ доступен на Docker Hub.

    Фронтенд #

    Фронтенд не распространяется как Open Source, но доступен для размещения на своем сервере. Вы можете найти Docker-образ на Docker Hub

    Установка #

    1. Вам нужно установить Docker и Docker Compose на своем компьютере, чтобы запустить Econumo.
    2. Создайте файл .env в корневом каталоге вашего проекта (см. пример ниже).
    3. Создайте файл docker-compose.yml в корневом каталоге вашего проекта с содержимым, перечисленным ниже.
    4. Запустите docker-compose up -d в корневом каталоге проекта.

    Переменные окружения #

    APP_ENV=prod
APP_SECRET=123 # Change this to a random string

DATABASE_USER=postgres
DATABASE_PASSWORD=postgres
DATABASE_NAME=db
DATABASE_HOST=db
DATABASE_PORT=5432
DATABASE_SSL_MODE=disable
DATABASE_URL=postgresql://${DATABASE_USER}:${DATABASE_PASSWORD}@${DATABASE_HOST}:${DATABASE_PORT}/${DATABASE_NAME}?sslmode=${DATABASE_SSL_MODE}

# If you want to use the system API to update currencies, you need to set this key
ECONUMO_SYSTEM_API_KEY=
# Change this to your base currency
ECONUMO_CURRENCY_BASE=USD
# Change this to false if you don't want to allow registration
ECONUMO_ALLOW_REGISTRATION=true
# Change this to your email address
ECONUMO_FROM_EMAIL=
# Change this to your email address
ECONUMO_REPLY_TO_EMAIL=
# Specify the secret and public key paths
JWT_SECRET_KEY=%kernel.project_dir%/config/jwt/private.pem
JWT_PUBLIC_KEY=%kernel.project_dir%/config/jwt/public.pem
# Change this to a random string
JWT_PASSPHRASE=123 

CORS_ALLOW_ORIGIN=*
MESSENGER_TRANSPORT_DSN=sync://
LOCK_DSN=semaphore
MAILER_DSN=null://null

NEW_RELIC_ENABLED=false
QASE_ENABLE=false

    Docker Compose File #

    version: '3.5'

services:
  app:
    image: econumo/app-frontend:latest
    restart: always
    depends_on:
      - api
    ports:
      - "8080:80"
    environment:
      - ECONUMO_CONFIG_API_URL="http://localhost:8082" # Change this to the URL of your API

  api:
    image: econumo/api-backend:latest
    restart: always
    env_file:
      - .env
    depends_on:
      - db
    ports:
      - "8082:80"

  db:
    image: postgres:13-alpine
    restart: always
    env_file:
      - .env
    volumes:
      - db-data:/var/lib/postgresql/data
    environment:
      - POSTGRES_HOST=${DATABASE_HOST}
      - POSTGRES_PORT=${DATABASE_PORT}
      - POSTGRES_USER=${DATABASE_USER}
      - POSTGRES_PASSWORD=${DATABASE_PASSWORD}
      - POSTGRES_DB=${DATABASE_NAME}

  db-data:
